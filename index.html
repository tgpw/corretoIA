SQL recomendado para v4

-- gabaritos (1 por disciplina)
drop table if exists public.gabaritos;
create table public.gabaritos (
  id bigint generated by default as identity primary key,
  created_at timestamptz default now(),
  nome text,
  modulo text,
  disciplina text,
  arquivo_nome text,
  arquivo_path text,
  arquivo_url text
);

-- sessoes / correções auditáveis
drop table if exists public.sessoes;
create table public.sessoes (
  id bigint generated by default as identity primary key,
  created_at timestamptz default now(),
  aluno text,
  modulo text,
  disciplina text,
  tipo text,
  nota numeric,
  arquivo text,
  data timestamptz,
  gabarito_id bigint,
  esperado text,
  descontos text,
  memoria_calculo text,
  feedback text
);
